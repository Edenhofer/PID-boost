	\begin{itemize}
		\item Construct a simple alternative to the `particleID' variables using Bayes with static priors
		\item Compare the static priors approach with the `particleID' one
		\item Introduce a Bayesian approach depending on chunked priors
		\item Visualize difference in effectiveness of chunked Bayes, simple Bayes and `particleID'
		\item Investigate possible alternatives to binning the transverse momentum $p_T$
	\end{itemize}

\begin{frame}
	\frametitle{Bayes with chunked Priors}
	\framesubtitle{Motivation for dependable a priori probabilities}

	\begin{columns}[T]
		\begin{column}{.5\textwidth}
			Reasoning
			\begin{itemize}
				\item A priori probabilities not independent of the measurements
				\item Different phase space for different measurements and hence different particle distribution
				\item Exclusion of un-physical quantities
			\end{itemize}
		\end{column}
		\begin{column}{.5\textwidth}
			\centering
			\includegraphics[width=\textwidth,height=0.8\textheight,keepaspectratio]{{{res/Chunked Bayesian Approach: Outliers Outside of 1.5 IQR on a Logarithmic Scale}}}
		\end{column}
	\end{columns}

	\vspace{2em}
	$\Rightarrow$ Adaptable priors dependant on measurements

\end{frame}
